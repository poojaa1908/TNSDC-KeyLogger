let
    Source = PreviousStepName,

    // Step 1: Add a record per row where list becomes a record like [hit1=..., hit2=..., ...]
    WithRecord = Table.AddColumn(Source, "ExpandedHits", each 
        let
            lst = if Value.Is([hits], type list) then [hits] else {},
            colNames = List.Transform({1..List.Count(lst)}, each "hit" & Text.From(_)),
            rec = Record.FromList(lst, colNames)
        in
            rec
    ),

    // Step 2: Expand all dynamically present fields in all rows
    Expanded = Table.ExpandRecordColumn(WithRecord, "ExpandedHits")
in
    Expanded
