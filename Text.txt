FilteredBots =
VALUES (
    FILTER (
        errors,
        // Custom logic to filter based on selected granularity and number
        errors[TimeColumn] >= TODAY() - SWITCH(
            SELECTEDVALUE(TimeGranularity[Granularity]),
            "Hours", SELECTEDVALUE(TimeRange[TimeRange]) / 24,
            "Days", SELECTEDVALUE(TimeRange[TimeRange]),
            "Weeks", SELECTEDVALUE(TimeRange[TimeRange]) * 7
        )
    )[bot_name]
)
