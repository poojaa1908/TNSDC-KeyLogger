SELECT 
    con.bot_conversationtranscriptid AS [BotName],
    con.conversationtranscriptid AS [ConversationID],
    con.createdon AS [CreatedOn],
    JSON_VALUE(activity.value, '$.type') AS [ValueType],
    JSON_VALUE(activity.value, '$.value') AS [Value]
FROM 
    conversationtranscript AS con
CROSS APPLY OPENJSON(con.content, '$.activities') AS activity
WITH (
    value NVARCHAR(MAX) AS JSON
)
