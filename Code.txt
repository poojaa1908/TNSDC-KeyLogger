avg-ratings = 
VAR SelectedVA = VALUES('virtual_assistant_table'[Virtual Assistant])
VAR FilteredRatings =
    CALCULATETABLE(
        'ratings',
        TREATAS(SelectedVA, 'ratings'[Virtual Assistant])  // use the actual column name used in ratings table
    )
VAR AvgScore =
    AVERAGEX(
        FILTER(FilteredRatings, NOT ISBLANK('ratings'[castScore])),
        'ratings'[castScore]
    )
RETURN
ROUND(COALESCE(AvgScore, 0), 2)q
